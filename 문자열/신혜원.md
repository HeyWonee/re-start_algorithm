## 튜플
1. replace, strip 으로 문자 제거
2. key:lambda x : len(x)

```
def solution(s):
    answer = []
    
#     replace와 rstrip 활용하여 문자 제거
    s = s.replace('{', '').rstrip('}')
#     }, 기준으로 split 하여 숫자 그룹 구분
    s = s.split('},')
    
    
#     s를 순회하며, ',' 기준으로 split + int로 변환
    for i in range(len(s)):
        s[i] = s[i].split(',')
        s[i] = list(map(int, s[i]))
    
#     원소 길이가 짧은 것 부터 정렬
    s.sort(key=lambda x : len(x))

#     순회하며, answer에 담기지 않았으면 append
    for i in range(len(s)):
        for j in range(len(s[i])):
            if s[i][j] not in answer:
                answer.append(s[i][j])
    
    
    return answer

```

## 짝지어 제거하기

```
from collections import deque

def solution(s):

    q = deque(list(s))
    stack = []
    
    while q:
        k = q.popleft()
        if stack:
            if stack[-1] == k:
                stack.pop()
            else:
                stack.append(k)
        else:
            stack.append(k)
    
    return 0 if stack else 1
```

def solution(s):
    number_transform = number_delete = 0
    N = len(s)

    while s != '1':
        #         replace
        s = s.replace('0', '')
        M = len(s)
        number_delete += (N - M)

        #         transoform

        # s = bin(M)[2:]
        k = M
        bi = ''
        while k != 1:
            bi = str(k % 2) + bi
            k //= 2

        else:
            bi = '1' + bi

        s = bi

        N = len(s)
        number_transform += 1

    return [number_transform, number_delete]
